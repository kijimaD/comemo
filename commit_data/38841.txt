    cmd/compile: reduce bits.Div64(0, lo, y) to 64 bit division
    
    With this change, these two functions generate identical code:
    
    func f(x uint64) (uint64, uint64) {
            return bits.Div64(0, x, 5)
    }
    
    func g(x uint64) (uint64, uint64) {
            return x / 5, x % 5
    }
