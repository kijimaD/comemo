    and then running:
    
    sed -i "" 's/\.U\.[SBXFC]val = /.U = /' *.go
    sed -i "" 's/\.U\.Sval/.U.\(string\)/g' *.go *.y
    sed -i "" 's/\.U\.Bval/.U.\(bool\)/g' *.go *.y
    sed -i "" 's/\.U\.Xval/.U.\(\*Mpint\)/g' *.go *.y
    sed -i "" 's/\.U\.Fval/.U.\(\*Mpflt\)/g' *.go *.y
    sed -i "" 's/\.U\.Cval/.U.\(\*Mpcplx\)/g' *.go *.y
