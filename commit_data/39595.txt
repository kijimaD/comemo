    cmd/compile: optimize arm64 comparison of x and 0.0 with "FCMP $(0.0), Fn"
    
    Code:
    func comp(x float64) bool {return x < 0}
    
    Previous version:
      FMOVD "".x(FP), F0
      FMOVD ZR, F1
      FCMPD F1, F0
      CSET  MI, R0
      MOVB  R0, "".~r1+8(FP)
      RET   (R30)
    
    Optimized version:
      FMOVD "".x(FP), F0
      FCMPD $(0.0), F0
      CSET  MI, R0
      MOVB  R0, "".~r1+8(FP)
      RET   (R30)
