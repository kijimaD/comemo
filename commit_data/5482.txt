+// func Fdim(x, y float64) float64
+TEXT 路Fdim(SB),7,$0
+	MOVSD x+0(FP), X0
+	SUBSD y+8(FP), X0
+	MOVSD $(0.0), X1
+	MAXSD X1, X0
+	MOVSD X0, r+16(FP)
+	RET
+
+// func Fmax(x, y float64) float64
+TEXT 路Fmax(SB),7,$0
+	MOVSD x+0(FP), X0
+	MAXSD y+8(FP), X0
+	MOVSD X0, r+16(FP)
+	RET
+
+// func Fmin(x, y float64) float64
+TEXT 路Fmin(SB),7,$0
+	MOVSD x+0(FP), X0
+	MINSD y+8(FP), X0
+	MOVSD X0, r+16(FP)
+	RET
